28  /**
29   * An {@link Appendable} sequence of strings or {@link StringHolder}s.
30   *
31   * @author Christian Kohlschütter
32   */
33  public class StringHolderSequence extends StringHolder implements Appendable {
34    final List<CharSequence> sequence;
35  
36    private boolean immutable = false;
37    private boolean cannotUseCache = false;
38    private Integer cachedHashCode = null;
39    private Integer cachedLength = null;

41    /**
42     * Constructs a new, empty {@link StringHolderSequence}.
43     */
44    public StringHolderSequence() {
45      this(10);
46    }

48    /**
49     * Constructs a new, empty {@link StringHolderSequence}.
50     *
51     * @param estimatedNumberOfAppends Estimated number of calls to {@link #append(Object)}, etc.
52     */
53    public StringHolderSequence(int estimatedNumberOfAppends) {
54      super(0);
55      sequence = new ArrayList<>(estimatedNumberOfAppends);
56    }

58    /**
59     * Checks if the to-be-appended {@link StringHolder} should be added as a {@link String} (with
60     * conversion via {@link StringHolder#toString()}) instead of adding it directly.
61     *
62     * This is false by default for all objects. Subclasses may override this selectively. If all
63     * objects should be converted to strings, use {@link StringOnlySequence}.
64     *
65     * @param sh The {@link StringHolder}.
66     * @return {@code true} if it should be appended as a string.
67     * @see StringOnlySequence
68     */
69    protected boolean needsStringConversion(StringHolder sh) {
70      return false;
71    }
72  
73    private void checkMutable() {
74      if (immutable) {
75        throw new IllegalStateException(
76            "Cannot append -- instance is marked as effectively-immutable");
77      }
78    }

80    @Override
81    public StringHolderSequence append(CharSequence csq, int start, int end) {
82      if (end == start) {
83        return this;
84      }
85  
86      if (csq instanceof String) {
87        append(((String) csq).substring(start, end));
88      } else if (csq instanceof StringBuilder) {
89        append(((StringBuilder) csq).substring(start, end));
90      } else if (csq instanceof StringBuffer) {
91        append(((StringBuffer) csq).substring(start, end));
92      } else {
93        append(csq.subSequence(start, end));
94      }
95      return this;
96    }

98    @Override
99    public StringHolderSequence append(char c) {
100     addSequence(String.valueOf(c));
101     return this;
102   }

104   /**
105    * Appends the given {@link String}, unless it is empty.
106    *
107    * @param s The string.
108    * @return This instance.
109    */
110   public StringHolderSequence append(String s) {
111     if (!s.isEmpty()) {
112       addSequence(CommonStrings.lookupIfPossible(s));
113     }
114     return this;
115   }

117   private void addSequence(String s) {
118     int len = s.length();
119     if (len == 0) {
120       return;
121     }
122 
123     checkMutable();
124 
125     uncache();
126     sequence.add(s);
127     resizeBy(len, len);
128   }
129 
130   /**
131    * Appends the given {@link StringHolder}, unless it is known to be empty.
132    *
133    * @param s The string.
134    * @return This instance.
135    */
136   public StringHolderSequence append(StringHolder s) {
137     if (s.isKnownEmpty()) {
138       return this;
139     } else if (s.isString() || needsStringConversion(s)) {
140       checkMutable();
141       addSequence(s.toString());
142       return this;
143     }
144 
145     if (!s.isEffectivelyImmutable()) {
146       cannotUseCache = true;
147     }
148 
149     uncache();
150     sequence.add(s);
151     resizeBy(s.getMinimumLength(), s.getExpectedLength());
152 
153     return this;
154   }

157    * Appends the given {@link CharSequence}, unless it is known to be empty.

162   public StringHolderSequence append(CharSequence s) {
163     if (!CharSequenceReleaseShim.isEmpty(s)) {
164       addSequence(CommonStrings.lookupIfPossible(String.valueOf(s)));
165     }
166     return this;
167   }

199   /**
200    * Appends the given object, unless it is known to be empty.
201    *
202    * The object
203    *
204    * @param s The string.
205    * @return This instance.
206    */
207   public StringHolderSequence append(Object s) {
208     if (s instanceof StringHolder) {
209       return append((StringHolder) s);
210     } else if (s instanceof String) {
211       return append((String) s);
212     } else if (s instanceof CharSequence) {
213       return append((CharSequence) s);
214     } else {
215       return append(String.valueOf(s));
216     }
217   }

219   @Override
220   protected int appendToAndReturnLengthDefaultImpl(Appendable out) throws IOException {
221     int len = 0;
222     for (Object obj : sequence) {
223       if (obj instanceof StringHolder) {
224         StringHolder holder = (StringHolder) obj;
225         if (holder.isKnownEmpty()) {
226           continue;
227         }
228 
229         len += holder.appendToAndReturnLength(out);
230       } else {
231         String s = (String) obj;
232         len += s.length();
233         out.append(s);
234       }
235     }
236     return len;
237   }

283   /**
284    * Appends to a list that only holds Strings or StringSuppliers, excluding
285    * {@link StringHolderSequence}s, whose contents are flattened to the list.
286    *
287    * @param flatList The list to append to.
288    * @return The minimum for the estimated length.
289    */
290   final int appendToFlatList(List<Object> flatList) {
291     int len = 0;
292     for (Object obj : sequence) {
293       if (obj instanceof StringHolder) {
294         StringHolder holder = (StringHolder) obj;
295         if (holder.isKnownEmpty()) {
296           continue;
297         }
298 
299         if (obj instanceof StringHolderSequence) {
300           len += ((StringHolderSequence) obj).appendToFlatList(flatList);
301         } else {
302           len += holder.getMinimumLength();
303           flatList.add(obj);
304         }
305       } else {
306         String s = (String) obj;
307         len += s.length();
308         flatList.add(s);
309       }
310     }
311     return len;
312   }
313 
314   @Override
315   protected String getString() {
316     StringBuilder sb = new StringBuilder(Math.max(16, getExpectedLength()));
317     int len = appendToAndReturnLength(sb);
318 
319     uncache();
320 
321     final String s;
322     sequence.clear();
323     if (len == 0) {
324       s = "";
325     } else {
326       s = sb.toString();
327       sequence.add(s);
328     }
329 
330     return s;
331   }
332 
333   @Override
334   protected Reader newReader() {
335     if (sequence.isEmpty()) {
336       return new StringReader("");
337     }
338 
339     final List<Object> flatList = new ArrayList<>(sequence.size());
340     int len = appendToFlatList(flatList);
341     if (len == 0) {
342       return new StringReader("");
343     }
344 
345     return new StringSequenceReader(flatList.iterator());
346   }

348   private static final class StringSequenceReader extends Reader {
349     private boolean closed = false;
350 
351     private String currentString = null;
352     private int currentPos;
353 
354     private final Iterator<Object> flatObjectIterator;
355 
356     private StringSequenceReader(Iterator<Object> flatObjectIterator) {
357       super();
358       this.flatObjectIterator = flatObjectIterator;
359     }
360 
361     private void ensureOpen() throws IOException {
362       if (closed) {
363         throw new IOException("Stream closed");
364       }
365     }
366 
367     @Override
368     public boolean ready() throws IOException {
369       ensureOpen();
370       return true;
371     }
372 
373     @Override
374     public void close() throws IOException {
375       closed = true;
376     }
377 
378     int ensureObject() {
379       if (currentString != null) {
380         int len = currentString.length();
381         if (currentPos >= len) {
382           currentString = null;
383           return ensureObject();
384         } else {
385           return len;
386         }
387       } else {
388         while (flatObjectIterator.hasNext()) {
389           currentString = flatObjectIterator.next().toString();
390           currentPos = 0;
391           int len = currentString.length();
392           if (len > 0) {
393             return len;
394           }
395         }
396         currentString = null;
397         return 0;
398       }
399     }
400 
401     @Override
402     public int read(char[] cbuf, int off, int len) throws IOException {
403       int currentLen = ensureObject();
404       if (currentLen == 0) {
405         return -1;
406       }
407 
408       len = Math.min(currentLen - currentPos, Math.min(cbuf.length - off, len));
409       currentString.getChars(currentPos, currentPos + len, cbuf, off);
410       currentPos += len;
411       return len;
412     }
413 
414     @Override
415     public int read() throws IOException {
416       int currentLen = ensureObject();
417       if (currentLen == 0) {
418         return -1;
419       }
420       return currentString.charAt(currentPos++);
421     }
422   }

429    */

431     return sequence.size();
432   }

434   /**
435    * Returns a simplified version of the contents of this sequence, if possible.
436    *
437    * <ol>
438    * <li>If the content is a string already, the string is returned.</li>
439    * <li>If there's no element stored, an empty string is returned.</li>
440    * <li>If there is only a single element stored, its content is returned as if
441    * {@link #asContent()} was called on that element.</li>
442    * </ol>
443    */
444   @Override
445   public Object asContent() {
446     if (isString()) {
447       return toString();
448     }
449     if (sequence.isEmpty()) {
450       return "";
451     } else if (sequence.size() == 1) {
452       Object obj = sequence.get(0);
453       if (obj instanceof String) {
454         return obj;
455       }
456       StringHolder sc = (StringHolder) obj;
457       return sc.asContent();
458     } else {
459       return this;
460     }
461   }

463   @Override
464   @SuppressWarnings("PMD.CognitiveComplexity")
465   public char charAt(int index) {
466     int offset = 0;
467 
468     if (index < 0) {
469       throw new IndexOutOfBoundsException();
470     }
471 
472     for (Object obj : sequence) {
473       final int len;
474 
475       if (obj instanceof StringHolder) {
476         StringHolder sh = (StringHolder) obj;
477         if (sh.isEmpty()) {
478           len = 0;
479         } else {
480           if (index == offset) {
481             return sh.charAt(index - offset);
482           }
483           len = sh.length();
484           if (index < offset + len) {
485             return sh.charAt(index - offset);
486           }
487         }
488       } else {
489         String s = (String) obj;
490         len = s.length();
491         if (index < offset + len) {
492           return s.charAt(index - offset);
493         }
494       }
495       offset += len;
496     }
497 
498     throw new IndexOutOfBoundsException();
499   }

501   @Override
502   public int hashCode() { // NOPMD.OverrideBothEqualsAndHashcode
503     if (cachedHashCode == null || cannotUseCache) {
504       cachedHashCode = updateHashCode(0);
505     }
506     return cachedHashCode;
507   }
508 
509   @Override
510   protected int updateHashCode(int h) {
511     if (isString()) {
512       return super.updateHashCode(h);
513     }
514 
515     for (CharSequence obj : sequence) {
516       h = updateHashCode(obj, h);
517     }
518 
519     return h;
520   }
521 
522   private static int updateHashCode(Object obj, int h) {
523     if (obj instanceof StringHolder) {
524       StringHolder sh = (StringHolder) obj;
525       if (!sh.isKnownEmpty()) {
526         h = sh.updateHashCode(h);
527       }
528     } else {
529       CharSequence s = (CharSequence) obj;
530       if (h == 0) {
531         return s.hashCode();
532       }
533       int len = s.length();
534       for (int i = 0; i < len; i++) {
535         h = 31 * h + s.charAt(i);
536       }
537     }
538 
539     return h;
540   }

542   @Override
543   protected boolean checkEquals(StringHolder sh) {
544     if (sh instanceof StringHolderSequence) {
545       StringHolderSequence shs = (StringHolderSequence) sh;
546       if (sequence.equals(shs.sequence)) {
547         return true;
548       }
549     }
550 
551     return super.checkEquals(sh);
552   }

554   @Override
555   protected void uncache() {
556     super.uncache();
557     cachedHashCode = null;
558     cachedLength = null;
559   }
560 
561   @Override
562   protected int computeLength() {
563     Integer length = cachedLength;
564     if (length == null || cannotUseCache) {
565       int len = 0;
566       for (CharSequence obj : sequence) {
567         len += obj.length();
568       }
569       cachedLength = length = len;
570     }
571     return length;
572   }

574   @Override
575   public boolean isEffectivelyImmutable() {
576     return immutable;
577   }
578 
579   @Override
580   public void markEffectivelyImmutable() {
581     if (immutable) {
582       return;
583     }
584     for (Object seq : sequence) {
585       if (!(seq instanceof StringHolder)) {
586         continue;
587       }
588       StringHolder sh = (StringHolder) seq;
589       sh.markEffectivelyImmutable();
590     }
591     immutable = true;
592   }
