13  public class StringsOnlyRenderTransformerTest {
14    @Test
15    public void testPrerenderStringHolder() throws Exception {
16      TemplateParser parser = new TemplateParser.Builder().withRenderSettings(
17          new RenderSettings.Builder().withRenderTransformer(StringsOnlyRenderTransformer
18              .getInstance()).build()).build();
19  
20      String json = "{\"array\" : [1,2,3] }";
21  
22      Template template = parser.parse("{% for item in array %}{{ item }}{% endfor %}");
23  
24      assertTrue(template.prerender(json) instanceof String,
25          "Prerendered result of a for-loop should be a String");
26    }
27  
28    @Test
29    public void testPrerenderString() throws Exception {
30      TemplateParser parser = new TemplateParser.Builder().withRenderSettings(
31          new RenderSettings.Builder().withRenderTransformer(StringsOnlyRenderTransformer
32              .getInstance()).build()).build();
33  
34      Template template = parser.parse("Hello World");
35  
36      assertTrue(template.prerender() instanceof String,
37          "Prerendered result of a simple string should be a String");
38    }
39  
40    @Test
41    public void testPrerenderLengthExceeded() throws Exception {
42      TemplateParser parser = new TemplateParser.Builder().withRenderSettings(
43          new RenderSettings.Builder().withRenderTransformer(StringsOnlyRenderTransformer
44              .getInstance()).build()).withProtectionSettings(new ProtectionSettings.Builder()
45                  .withMaxSizeRenderedString(2).build()).build();
46  
47      String json = "{\"array\" : [1,2,3] }";
48      Template template = parser.parse("{% for item in array %}{{ item }}{% endfor %}");
49  
50      assertThrows(Exception.class, () -> template.prerender(json),
51          "Exception should be thrown because string exceeds limit");
52    }
